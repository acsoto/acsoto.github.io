class Account{
    int id;
    int balance;
public:
    Account &operator+(int n);
    //++a
    Account& operator++();
    //a++
    Account operator++(int);
};

Account &Account::operator+(int n) {
    this->balance += n;
}

Account& Account::operator++() {
    this->balance++;
    return *this; 
}

Account Account::operator++(int) {
    Account old = *this;
    this->balance++;
    return old;
}

int main()
{
    Account a;
    a = a+100;
}

行为串联：当两个类别家族因为行为共性发生关系时，用抽象类来关联两个类别家族

template class
//template class 模板类 collection in Java
template<class T>
class Stack {
    T pool[100];
    int top;
public:
    //内联函数 inline function
    //top=0
    Stack() : top(0) {
    }

    void push(T v) {
        pool[top++] = v;
    }

    T pop() {
        return pool[--top];
    }
};
int main() {
    Stack<int> s1;
    Stack<double> s2;
}
